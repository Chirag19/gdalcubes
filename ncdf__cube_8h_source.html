65  void select_bands(std::vector<std::string> bands) {
66  _band_selection.clear();
67  if (bands.empty()) {
68  _bands = _orig_bands;
69  }
70  else {
71  band_collection bands_new;
72  for (uint16_t i=0; i< bands.size(); ++i) {
73  if (_orig_bands.has(bands[i])) {
74  bands_new.add(_orig_bands.get(bands[i]));
75  _band_selection.push_back(bands[i]);
76  }
77  else {
78  GCBS_WARN("Data cube has no band with name '" + bands[i] + "'; band will be skipped");
79  }
80  }
81  if (bands_new.count() > 0) {
82  _bands = bands_new;
83  }
84  else {
85  _bands = _orig_bands;
86  }
87  }
88  }
