66  static duration from_string(std::string s) {
67  boost::regex rexp("P(T?)([0-9]+)([YMWDHS])");
68 
69  boost::cmatch res;
70  if (!boost::regex_match(s.c_str(), res, rexp)) {
71  throw std::string("ERROR in duration::from_string(): cannot derive date interval");
72  }
73  duration d;
74  d.dt_interval = std::stoi(res[2]);
75  if (!res[1].str().empty()) {
76  if (res[3] == "H")
77  d.dt_unit = HOUR;
78  else if (res[3] == "M")
79  d.dt_unit = MINUTE;
80  else if (res[3] == "S")
81  d.dt_unit = SECOND;
82  } else {
83  if (res[3] == "Y")
84  d.dt_unit = YEAR;
85  else if (res[3] == "M")
86  d.dt_unit = MONTH;
87  else if (res[3] == "W")
88  d.dt_unit = WEEK;
89  else if (res[3] == "D")
90  d.dt_unit = DAY;
91  }
92  return d;
93  }

214 
215  boost::cmatch res;
216  if (!boost::regex_match(s.c_str(), res, regex1)) {
217  throw std::string("ERROR in datetime::from_string(): cannot derive datetime from string");
218  } else {

222  if (i == 2) {

224  out._unit = YEAR;
225  } else if (i == 3) {
226  out._p = boost::posix_time::ptime(boost::gregorian::date(std::stoi(res[1].str()), std::stoi(res[2].str()), 1), boost::posix_time::time_duration(0, 0, 0));
227  out._unit = MONTH;
228  } else if (i == 4) {
229  out._p = boost::posix_time::ptime(boost::gregorian::date(std::stoi(res[1].str()), std::stoi(res[2].str()), std::stoi(res[3].str())), boost::posix_time::time_duration(0, 0, 0));

254  friend duration operator-(const datetime& l,
255  const datetime& r) {
256  duration out;
257  out.dt_unit = std::max(l.unit(), r.unit()); // TODO: warning if not the same unit
258  switch (out.dt_unit) {
259  case SECOND:
260  out.dt_interval = (l._p - r._p).total_seconds();
261  break;
262  case MINUTE:
263  out.dt_interval = (l._p - r._p).total_seconds() / 60;
264  break;
265  case HOUR:
266  out.dt_interval = (l._p - r._p).total_seconds() / (3600);
267  break;
268  case DAY:
269  out.dt_interval = (l._p.date() - r._p.date()).days();
270  break;
271  case WEEK:
272  out.dt_interval = (l._p.date() - r._p.date()).days() / 7;
273  break;
274  case MONTH: {
275  int yd = l._p.date().year() - r._p.date().year();
276  if (std::abs(yd) >= 1) {
277  out.dt_interval = yd * 12 + (l._p.date().month() - r._p.date().month());
278  } else if (yd == 0) {
279  out.dt_interval = (l._p.date().month() - r._p.date().month());
280  }
281  break;
282  }
283  case YEAR:
284  out.dt_interval = l._p.date().year() - r._p.date().year();
285  break;
286  }
287  return out;
288  }

298  case MINUTE:
299  return (l._p.date() == r._p.date() &&
300  l._p.time_of_day().hours() == r._p.time_of_day().hours() &&
301  l._p.time_of_day().minutes() == r._p.time_of_day().minutes());
302  case HOUR:
303  return (l._p.date() == r._p.date() &&
304  l._p.time_of_day().hours() == r._p.time_of_day().hours());

307  case WEEK:
308  // TODO: what if the same week overlaps two years?
309  return (l._p.date().year() == r._p.date().year() && l._p.date().week_number() == r._p.date().week_number());
310  case MONTH:
311  return (l._p.date().year() == r._p.date().year() && l._p.date().month() == r._p.date().month());
312 
313  case YEAR:

358  friend datetime operator+(datetime l, const duration& r) {
359  datetime out(l._p);
360  switch (r.dt_unit) {
361  case SECOND:
362  out._p += boost::posix_time::seconds(r.dt_interval);
363  break;
364  case MINUTE:
365  out._p += boost::posix_time::minutes(r.dt_interval);
366  break;
367  case HOUR:
368  out._p += boost::posix_time::hours(r.dt_interval);
369  break;
370  case DAY:
371  out = boost::posix_time::ptime(out._p.date() + boost::gregorian::days(r.dt_interval)); // ignore time
372  break;
373  case WEEK:
374  out = boost::posix_time::ptime(out._p.date() + boost::gregorian::days(r.dt_interval * 7)); // ignore time
375  break;
376  case MONTH:
377  out = boost::posix_time::ptime(boost::gregorian::date(out._p.date().year() + (long)(r.dt_interval / 12), out._p.date().month() + r.dt_interval % 12, 1)); // ignore time and day
378  break;
379  case YEAR:
380  out = boost::posix_time::ptime(boost::gregorian::date(out._p.date().year() + r.dt_interval, 1, 1)); // ignore time, day, and month
381  break;
382  }
383  return out;
384  }
