157  chunkid_t find_chunk_that_contains(coords_st p) const {
158  uint32_t cumprod = 1;
159  chunkid_t id = 0;
160 
161  // 1. Convert map coordinates to view-based coordinates
162  coords_nd<uint32_t, 3> s = _st_ref->cube_coords(p);
163 
164  // 2. Find out which chunk contains the integer view coordinates
165  id += cumprod * (s[2] / _chunk_size[2]);
166  cumprod *= (uint32_t)std::ceil(((double)_st_ref->nx()) / ((double)_chunk_size[2]));
167  id += cumprod * (s[1] / _chunk_size[1]);
168  cumprod *= (uint32_t)std::ceil(((double)(_st_ref->ny()) / ((double)_chunk_size[1])));
169  id += cumprod * (s[0] / _chunk_size[0]);
170  //cumprod *= (uint32_t)std::ceil(((double)(_c->view().nt()) / ((double)_size_t);
171 
172  return id;

176  cube_coordinate_tyx out_vcoords_low;
177  cube_coordinate_tyx out_vcoords_high;
178 
179  out_vcoords_low[0] = c[0] * _chunk_size[0];
180  out_vcoords_low[1] = c[1] * _chunk_size[1];
181  out_vcoords_low[2] = c[2] * _chunk_size[2];
182  out_vcoords_high[0] = out_vcoords_low[0] + _chunk_size[0] - 1;
183  out_vcoords_high[1] = out_vcoords_low[1] + _chunk_size[1] - 1;
184  out_vcoords_high[2] = out_vcoords_low[2] + _chunk_size[2] - 1;
185 
186  // Shrink to view
187  if (out_vcoords_high[0] < 0)
188  out_vcoords_high[0] = 0;
189  else if (out_vcoords_high[0] >= _st_ref->nt())
190  out_vcoords_high[0] = _st_ref->nt() - 1;
191  if (out_vcoords_low[0] < 0)
192  out_vcoords_low[0] = 0;
193  else if (out_vcoords_low[0] >= _st_ref->nt())
194  out_vcoords_low[0] = _st_ref->nt() - 1;
195 
196  if (out_vcoords_high[1] < 0)
197  out_vcoords_high[1] = 0;
198  else if (out_vcoords_high[1] >= _st_ref->ny())
199  out_vcoords_high[1] = _st_ref->ny() - 1;
200  if (out_vcoords_low[1] < 0)
201  out_vcoords_low[1] = 0;
202  else if (out_vcoords_low[1] >= _st_ref->ny())

220  bounds_nd<uint32_t, 3> chunk_limits(chunkid_t id) const {
221  coords_nd<uint32_t, 3> out_vcoords_low;
222  coords_nd<uint32_t, 3> out_vcoords_high;
223 
224  // Transform to global coordinates based on chunk id
225  uint32_t cumprod = 1;
226  int32_t n;
227 
228  n = (uint32_t)std::ceil(((double)(_st_ref->nx())) / ((double)_chunk_size[2]));
229  out_vcoords_low[2] = (id / cumprod) % n;
230  out_vcoords_low[2] *= _chunk_size[2];
231  out_vcoords_high[2] = out_vcoords_low[2] + _chunk_size[2] - 1;
232  cumprod *= n;
233 
234  n = (uint32_t)std::ceil(((double)(_st_ref->ny())) / ((double)_chunk_size[1]));
235  out_vcoords_low[1] = (id / cumprod) % n;
236  out_vcoords_low[1] *= _chunk_size[1];
237  out_vcoords_high[1] = out_vcoords_low[1] + _chunk_size[1] - 1;
238  cumprod *= n;
239 
240  n = (uint32_t)std::ceil(((double)(_st_ref->nt())) / ((double)_chunk_size[0]));
241  out_vcoords_low[0] = (id / cumprod) % n;
242  out_vcoords_low[0] *= _chunk_size[0];
243  out_vcoords_high[0] = out_vcoords_low[0] + _chunk_size[0] - 1;
244  cumprod *= n;
245 
246  // Shrink to view
247  if (out_vcoords_high[0] < 0)
248  out_vcoords_high[0] = 0;
249  else if (out_vcoords_high[0] >= _st_ref->nt())
250  out_vcoords_high[0] = _st_ref->nt() - 1;
251  if (out_vcoords_low[0] < 0)
252  out_vcoords_low[0] = 0;
253  else if (out_vcoords_low[0] >= _st_ref->nt())
254  out_vcoords_low[0] = _st_ref->nt() - 1;
255 
256  if (out_vcoords_high[1] < 0)
257  out_vcoords_high[1] = 0;

259  out_vcoords_high[1] = _st_ref->ny() - 1;
260  if (out_vcoords_low[1] < 0)
261  out_vcoords_low[1] = 0;
262  else if (out_vcoords_low[1] >= _st_ref->ny())
263  out_vcoords_low[1] = _st_ref->ny() - 1;
264 
265  if (out_vcoords_high[2] < 0)
